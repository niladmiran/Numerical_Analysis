\begin{enumerate}
  \item We have $f(x^*)=0$ and $f''(x^*)=0$ at a point $x^*\in\mathbb{R}$, and 
  \item we have $f'(x^*)\neq 0$.

\begin{equation*}
  |x^{k+1}-x^k|\leq \gamma|x^k-x^*|^3
